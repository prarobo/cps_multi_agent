1function [dist] = dijkstraGameEff_mod(TM_old, F,turn)
 2%Modified dijkstra's algorithm for a game transition system
 3%
 4%Inputs are a transition matrix (can be sparse), destination node, and
 5%whose turn it is (1 is robot's turn, 0 is adversary's turn)
 6%
 7%Written by Kevin Leahy March 2015
 8
 9TM = zeros([size(TM_old,1) size(TM_old,2)]+1);
10TM(1:end-1,1:end-1) = TM_old;
11% TM(end+1,:) = zeros(1,length(TM));%add row and column of zeros to add virtual end node
12% TM(:,end+1) = zeros(length(TM),1);
13TM(F,end) = 1; %add transitions to virtual node
14n = length(TM);
15dist = ones(n,1)*Inf;
16dest = n; %make virtual node destination
17dist(dest) = 0; %set dist to virtual node as 0
18dist(F) = 0; %set dist to final states as 0
19Q = 1:n-1; %consider all but final node
20qInd = length(Q);
21% Q = setdiff(Q,F); %remove final states from Q
22
23% turnD = turn(dest); %is destination node robot turn?
24
25while( qInd ~= 0) %begin dijkstra
26    [val, qu] = min(dist(Q(1:qInd)));
27    u = Q(qu); % index of the vertex
28    if( isinf(dist(u) ))
29        break;
30    end
31    Q(qu:end-1) = Q(qu+1:end);
32    Q(end) = Inf;
33    qInd = qInd-1;
34    % Q = setdiff(Q, u);
35    % all transitions to u TM(:,u);
36    ind = (find(TM(:,u)==1))';
37    for j=1:length(ind)
38        i = ind(j);
39        if turn(i) == 1 %robot's turn -- use regular dijkstra
40            if dist(i) > (val + TM(i, u)) %&& (TM(i,u)~=0)
41                dist(i) = val + TM(i, u);
42            end
43        else %environment's turn
44            if ~isempty(find(TM(i,:)==1, 1)) && ~ismember(i,F)
45                dist(i) = max(dist(TM(i,:)==1))+1; %maximum energy for next states
46            end
47        end
48    end
49end
50% dist(dest) = 0; %manually make distance to this state 0
51dist = dist(1:end-1);
52% dist = dist - 1; %remove distance to virtual node, and keep distance to real nodes
53end
